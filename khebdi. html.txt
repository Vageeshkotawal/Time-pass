    <script>
        const particleContainer = document.getElementById('particle-container');
        let particleInterval;
        let activeMode = 'roses'; // tracks what falls: 'roses', 'celebration', or 'angry'

        // Creates falling elements based on the current active page theme
        function spawnParticle() {
            const item = document.createElement('div');
            item.classList.add('falling-item');
            
            let pool = [];
            if (activeMode === 'roses') {
                pool = ['🌹', '💛', '🌸', '❤️', '💗'];
            } else if (activeMode === 'celebration') {
                pool = ['🎉', '💥', '🥳', '✨', '❤️', '🌹', '🎈'];
            } else if (activeMode === 'angry') {
                pool = ['🤬', '😡', '🔥', '👿', '👊', '💀'];
            }

            item.innerHTML = pool[Math.floor(Math.random() * pool.length)];
            item.style.left = Math.random() * 100 + 'vw';
            item.style.fontSize = Math.random() * 20 + 20 + 'px';
            item.style.animationDuration = Math.random() * 3 + 2 + 's';
            
            particleContainer.appendChild(item);
            setTimeout(() => item.remove(), 5000);
        }

        // Initialize background particle generation
        function startParticles(mode) {
            clearInterval(particleInterval);
            activeMode = mode;
            // Faster generation for celebration/angry rain
            const rate = mode === 'roses' ? 250 : 80; 
            particleInterval = setInterval(spawnParticle, rate);
        }

        // Setup starting screen
        startParticles('roses');

        // Navigation Controls
        function goToPage2() {
            document.getElementById('page1').classList.add('hidden');
            document.getElementById('page2').classList.remove('hidden');
        }

        // Haa Route
        function goToSuccess() {
            document.getElementById('page2').classList.add('hidden');
            document.getElementById('page3').classList.remove('hidden');
            startParticles('celebration');
        }

        // Naah Route
        function goToAngry() {
            document.getElementById('page2').classList.add('hidden');
            document.body.classList.add('angry-mode');
            
            const p4 = document.getElementById('page4');
            p4.classList.remove('hidden');
            p4.classList.add('angry-card');
            
            startParticles('angry');
        }

        // Back from Naah Route
        function returnToProposal() {
            document.getElementById('page4').classList.add('hidden');
            document.getElementById('page4').classList.remove('angry-card');
            document.body.classList.remove('angry-mode');
            
            document.getElementById('page2').classList.remove('hidden');
            startParticles('roses');
        }

        // Back to Start from Love Page
        function returnToHome() {
            // Clear out any stray celebratory particles lingering on the screen
            particleContainer.innerHTML = '';
            
            document.getElementById('page3').classList.add('hidden');
            document.getElementById('page1').classList.remove('hidden');
            startParticles('roses');
        }
    </script>
